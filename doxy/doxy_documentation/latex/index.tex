{\itshape This project reads and saves a textfile in a generated hashtable and censores words depending on requested user input.} ~\newline
 ~\newline




\DoxyHorRuler{0}
\hypertarget{index_autotoc_md8}{}\doxysubsection{About The Project}\label{index_autotoc_md8}
{\bfseries{1 Part\+: (S\+O\+L\+V\+ED)~\newline
}}
\begin{DoxyItemize}
\item Open a text file and read the text row by row.~\newline

\item Separate words by these characters (\char`\"{} \char`\"{}, \char`\"{}.\char`\"{}, \char`\"{}\textbackslash{}n\char`\"{})~\newline

\item Store the values in a hashtable using dynamic linked lists.~\newline

\item Use the sum of character values as value to be hashed. E.\+g. Cat = (67+97+116) \% 23~\newline

\item Words shall be stored in dynamically allocated memory~\newline

\item Print out the hash in a readable way. E.\+g. max 10 words per row.~\newline

\end{DoxyItemize}

{\bfseries{2 Part\+: (S\+O\+L\+V\+ED)~\newline
}}
\begin{DoxyItemize}
\item The occurrence of words shall be counted instead of multiple entries in the list. ~\newline

\item Add additional word separators\+:~\newline

\item semicolon (;) ~\newline

\item colon (\+:)~\newline

\item comma (,)~\newline

\item question mark (?)~\newline

\item tab (\textbackslash{}t)~\newline

\end{DoxyItemize}

{\bfseries{3 Part\+: (S\+O\+L\+V\+ED)~\newline
}}
\begin{DoxyItemize}
\item The lists shall always retain alphabetic order.
\end{DoxyItemize}

{\bfseries{4 Part\+: (S\+O\+L\+V\+ED)~\newline
}}
\begin{DoxyItemize}
\item Give the user a choice to select one bucket to output separately.
\item Print out the entries in the Bucket.
\item Let the user select one or more buckets and remove all words from the text which do not match to these selected buckets and write it to a new text file.
\end{DoxyItemize}

{\bfseries{5 Part\+: (S\+O\+L\+V\+ED)~\newline
}}
\begin{DoxyItemize}
\item Let the user select one or more buckets and censor all words from the text which match to these selected buckets and write it to a new text file.
\item All word separators shall be copied to the new file in the correct places.
\end{DoxyItemize}

{\bfseries{6 Part\+: (S\+O\+L\+V\+ED)~\newline
}}
\begin{DoxyItemize}
\item Store the hash in a binary file
\item Read and populate the hash from a file and append/count new words into it when the user wants to do this.
\end{DoxyItemize}

{\bfseries{7 Part\+: (S\+O\+L\+V\+ED)~\newline
}}
\begin{DoxyItemize}
\item The program arguments are handled by getopt(3).
\end{DoxyItemize}
